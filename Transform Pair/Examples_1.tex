\subsection*{Order $n = 3$ problems}
\subsubsection*{Problem 1}
Let $x_0 = 0 < x_1  = 0.25 < x_2 = 0.7 < x_3 = 1$ be the partition of $[0,1],$ and consider the following problem:
\begin{align*}
\partial_t q - \partial_{xxx} q &= 0 &&\text{(PDE)} \\
q(x, 0) &= f(x) &&\text{(IC)} \\
